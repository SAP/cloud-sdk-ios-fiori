            tags: ["I am selected", "PM01", "103-Repair", "tag 4", "tag 5", "tag 6"],

            descriptionText: "Customer noticed that the transformer started",
            status: "High",
            substatus: "Scheduled",
            detailContent: {
                HeaderChart(title: {
                   Text("Temperature")
                }, subtitle: {
                   Text("20 min ago")

                })
